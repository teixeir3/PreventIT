def create
  @practice = Practice.new(params[:practice])
  @user = @practice.doctors.build(params[:user])
  @user.is_doctor = true
  @user.alert_setting.build

  if @practice.save
    sign_in(@user)
    redirect_to doctor_url(@user)
  else
    flash.now[:errors] = @practice.errors.full_messages
    render :doctor_new
  end
end
